\begin{frame}{Setup}
  \Large
  \textbf{Question:} if a persona vector changes model behavior, does that direction have a coherent steering radius?

  \vspace{1em}
  \normalsize
  \begin{itemize}
    \item Anchor: Anthropic-style persona vectors.
    \item Model: \qwen, mainly 7B.
    \item Traits: hallucination, evil, sycophancy.
    \item Control: same-normalization random vectors.
  \end{itemize}

  \vspace{0.5em}
  \begin{block}{Key idea}
    I sweep normalized steering strength $\beta$ and ask when judged coherence falls below 70.
  \end{block}
\end{frame}

\begin{frame}{Method}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.53\textwidth}
      \textbf{Vector extraction}
      \[
      v_{\text{trait}}[\ell]
      =
      \mathbb{E}[z_{\text{positive}}[\ell]]
      -
      \mathbb{E}[z_{\text{negative}}[\ell]]
      \]
      \begin{itemize}
        \item Positive/negative trait instructions.
        \item Mean over assistant response-token activations.
        \item Retain contrastive pairs using an LLM judge.
      \end{itemize}
    \end{column}
    \begin{column}{0.44\textwidth}
      \textbf{Response steering}
      \[
      h[:, -1, :] \leftarrow h[:, -1, :] + \beta \sigma \hat v
      \]
      \begin{itemize}
        \item Layer 20 residual stream.
        \item $\sigma$ = baseline projection std.
        \item Judge trait score and coherence.
      \end{itemize}
    \end{column}
  \end{columns}

  \vspace{0.7em}
  \begin{block}{Metric}
    $\diam$ = largest contiguous beta interval containing 0 with mean coherence $\ge 70$.
  \end{block}
\end{frame}

\begin{frame}{Main Result: Qwen2.5-7B}
  \begin{table}
    \centering
    \large
    \begin{tabular}{lccc}
      \toprule
      Direction & Coherent interval & Diameter & Ratio vs random \\
      \midrule
      hallucination vector & $[-8, 8]$ & 16 & 0.125 \\
      evil vector & $[-16, 8]$ & 24 & 0.188 \\
      sycophancy vector & $[-24, 16]$ & 40 & 0.313 \\
      random controls & $[-64, 64]$ & 128 & 1.000 \\
      \bottomrule
    \end{tabular}
  \end{table}

  \vspace{1em}
  \Large
  \textbf{Result:} trait directions break coherence much earlier than random directions.
\end{frame}

\begin{frame}{What This Means}
  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.48\textwidth}
      \textbf{Not just activation noise}
      \begin{itemize}
        \item Random vectors were coherent through $\pm 64$.
        \item Trait vectors failed much earlier.
        \item Same normalization procedure for both.
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Trait directions are structured}
      \begin{itemize}
        \item Strong beta increases trait expression.
        \item Past a point, outputs become repetitive or incoherent.
        \item Positive and negative directions are asymmetric.
      \end{itemize}
    \end{column}
  \end{columns}

  \vspace{1em}
  \begin{block}{Interpretation}
    Learned persona vectors look like behaviorally meaningful, high-sensitivity directions in residual space.
  \end{block}
\end{frame}

\begin{frame}{Cross-Model Check: Qwen2.5-3B}
  \begin{table}
    \centering
    \large
    \begin{tabular}{lccc}
      \toprule
      Direction & Qwen7 diameter & Qwen3 diameter & Same pattern? \\
      \midrule
      hallucination vector & 16 & 32 & yes \\
      evil vector & 24 & 48 & yes \\
      sycophancy vector & 40 & 48 & yes \\
      random controls & 128 & 128 & yes \\
      \bottomrule
    \end{tabular}
  \end{table}

  \vspace{1em}
  \begin{itemize}
    \item Qualitative pattern transfers across model size.
    \item Exact radii differ, so normalized diameter is not automatically model-invariant.
    \item Hallucination is the tightest direction in both models.
  \end{itemize}
\end{frame}

\begin{frame}{Fine Boundary Checks}
  \begin{table}
    \centering
    \large
    \begin{tabular}{lcc}
      \toprule
      Trait & Coarse interval & Fine-sweep update \\
      \midrule
      hallucination & $[-8, 8]$ & fails by $\pm 11$ \\
      evil & $[-16, 8]$ & approx. $[-19, 14]$ \\
      sycophancy & $[-24, 16]$ & near $-24$, fails between $16$ and $19$ \\
      \bottomrule
    \end{tabular}
  \end{table}

  \vspace{1em}
  \begin{itemize}
    \item Main diameters are grid-based estimates, not exact continuous radii.
    \item Fine sweeps probe around boundary points.
    \item Fine grids omit 0, so they refine boundaries but do not redefine the official metric.
  \end{itemize}
\end{frame}

\begin{frame}{Takeaways}
  \begin{enumerate}
    \item Persona-vector steering has an empirical coherence radius.
    \item Trait vectors are less tolerant than random controls under the same normalization.
    \item Different traits have different radii; hallucination is tightest here.
    \item The phenomenon appears in both Qwen2.5-7B and Qwen2.5-3B.
  \end{enumerate}

  \vspace{1em}
  \begin{block}{Limitations}
    LLM judge metrics, small retained-pair counts, one layer/hook point, generated artifacts, and no mechanistic proof yet.
  \end{block}
\end{frame}

\begin{frame}{Backup: Geometry}
  \begin{itemize}
    \item Within-model cosine similarities among the three core vectors were mostly near orthogonal.
    \item Qwen7 had weak structure:
      \[
      \cos(\text{evil}, \text{hallucination}) = 0.145,\quad
      \cos(\text{evil}, \text{sycophancy}) = -0.231
      \]
    \item Qwen3 was flatter:
      \[
      \cos(\text{evil}, \text{hallucination}) = 0.055,\quad
      \cos(\text{evil}, \text{sycophancy}) = 0.015
      \]
    \item Cross-model vector cosines are not meaningful directly because residual spaces are not aligned.
  \end{itemize}
\end{frame}

\begin{frame}{Backup: Failure Modes}
  Coherence failures usually looked like:

  \vspace{0.5em}
  \begin{itemize}
    \item repetitive phrasing or loops,
    \item abrupt loss of relevance,
    \item overexpression of the target trait,
    \item strange formatting or very short degenerate completions.
  \end{itemize}

  \vspace{1em}
  This is why the main metric uses behavioral coherence rather than just activation projection.
\end{frame}
